function startCamera() { navigator.mediaDevices.getUserMedia({ video: true }) .then(stream => { video.srcObject = stream; video.onloadedmetadata = () => { canvas.width = video.videoWidth; canvas.height = video.videoHeight; drawLoop(); }; }) .catch(err => alert('Ошибка доступа к камере')); } function drawLoop() { requestAnimationFrame(drawLoop); ctx.clearRect(0, 0, canvas.width, canvas.height); drawGrid(); } function drawGrid() { const heightCm = parseInt(document.getElementById('heightInput').value); if (!heightCm) return; const pixelsPerCm = canvas.height / heightCm; const centerX = canvas.width / 2; ctx.strokeStyle = 'rgba(255,255,255,0.4)'; ctx.lineWidth = 1; // Вертикальная ось ctx.beginPath(); ctx.moveTo(centerX, 0); ctx.lineTo(centerX, canvas.height); ctx.stroke(); // Горизонтальные линии каждые 10 см for (let cm = 0; cm <= heightCm; cm += 10) { const y = canvas.height - cm * pixelsPerCm; ctx.beginPath(); ctx.moveTo(0, y); ctx.lineTo(canvas.width, y); ctx.stroke(); } }
